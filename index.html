function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = sheet.getDataRange().getDisplayValues(); return ContentService.createTextOutput(JSON.stringify({status: "success", data: data})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // ดึงคอลัมน์หัวตารางทั้งหมด และตัดช่องว่างส่วนเกินออกเพื่อความแม่นยำ 100% var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(function(h) { return String(h).trim(); }); var data = JSON.parse(e.postData.contents); var action = data.action; var uidColIndex = headers.indexOf("UID"); var rowIndex = -1; // ค้นหาตำแหน่งบรรทัดของข้อมูลเดิมใน Google Sheet (กรณีแก้ไข หรือต้องการลบ) if (action === "update" || action === "delete") { var lastRow = sheet.getLastRow(); if (lastRow > 1 && uidColIndex > -1) { var searchData = sheet.getRange(2, uidColIndex + 1, lastRow - 1, 1).getValues(); for (var j = 0; j < searchData.length; j++) { if (String(searchData[j][0]).trim() === String(data.uid).trim()) { rowIndex = j + 2; break; } } } } // ------------------------------------------------------------- // ระบบ 1: สั่งลบข้อมูลจริงออกจากแถวใน Google Sheet // ------------------------------------------------------------- if (action === "delete") { if (rowIndex > -1) { sheet.deleteRow(rowIndex); return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "ลบข้อมูลสำเร็จ"})) .setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": "ไม่พบ UID ดังกล่าว"})) .setMimeType(ContentService.MimeType.JSON); } } // ------------------------------------------------------------- // ระบบ 2: จัดระเบียบชุดข้อมูลเพื่อเพิ่มใหม่ (Add) หรือบันทึกทับ (Update) // ------------------------------------------------------------- var obj = data.datasetObj; var oldRow = []; if (rowIndex > -1) { oldRow = sheet.getRange(rowIndex, 1, 1, headers.length).getValues()[0]; } var newRow = []; for (var i = 0; i < headers.length; i++) { if (headers[i] === "UID") { newRow.push(data.uid); } else if (headers[i] === "วันที่") { newRow.push(data.date); } else if (headers[i] === "เวร") { newRow.push(data.shift); } else if (obj && obj[headers[i]] !== undefined) { newRow.push(obj[headers[i]]); } else { // หากฝั่งผู้ใช้ไม่ได้อัปเดตฟิลด์ใดฟิลด์หนึ่ง ให้ใส่ค่าข้อมูลเดิมกลับคืน ป้องกันค่าตกหล่นหาย if (rowIndex > -1 && oldRow[i] !== undefined) { newRow.push(oldRow[i]); } else { newRow.push(""); } } } // บันทึกการเปลี่ยนแปลงลง Google Sheets if (rowIndex > -1 && action === "update") { sheet.getRange(rowIndex, 1, 1, newRow.length).setValues([newRow]); // สั่งบันทึกข้อมูลทับแถวเดิม } else { sheet.appendRow(newRow); // หากหาแถวเก่าไม่เจอ หรือเป็นข้อมูลใหม่ ให้เพิ่มเป็นแถวใหม่ด้านล่างสุดแทน } return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
